<script>
let moodData = JSON.parse(localStorage.getItem("moodEntries")) || [];

function saveEntry() {
    let role = document.getElementById("role").value;
    let mood = document.getElementById("mood").value;
    let reflection = document.getElementById("reflection").value;
    let date = new Date().toLocaleDateString();

    // Role-based feature restriction
    if (role !== "admin" && moodData.length >= 10) {
        alert("Upgrade to admin for unlimited entries.");
        return;
    }

    if (mood && reflection) {
        moodData.push({ date, mood, reflection });
        localStorage.setItem("moodEntries", JSON.stringify(moodData));
        updateChart();
        document.getElementById("reflection").value = "";
        alert("Entry saved successfully!");
    } else {
        alert("Please select a mood and write your reflection.");
    }
}

function updateChart() {
    let moodCounts = {};
    moodData.forEach(entry => {
        moodCounts[entry.mood] = (moodCounts[entry.mood] || 0) + 1;
    });

    let labels = Object.keys(moodCounts);
    let counts = Object.values(moodCounts);

    new Chart(document.getElementById("moodChart"), {
        type: 'bar',
        data: {
            labels: labels,
            datasets: [{
                label: 'Mood Frequency',
                data: counts,
                backgroundColor: ['#ffcc00', '#3399ff', '#cccccc', '#66ff66', '#ff6666']
            }]
        }
    });
}

updateChart();
